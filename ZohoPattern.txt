import java.io.*;
public class Zoho1 
{
	public static void main(String[] args) 
	{
	    int i,j;
	    for(i=5;i>0;i--)
	    {
	        for(j=5;j>0;j--)
	        {
	            if(j>i) 
	            {
	                System.out.print(j);
	            }
	            else
	            {
	                System.out.print(i);
	            }
	        }
	        System.out.println();
	    }
	}	
}
